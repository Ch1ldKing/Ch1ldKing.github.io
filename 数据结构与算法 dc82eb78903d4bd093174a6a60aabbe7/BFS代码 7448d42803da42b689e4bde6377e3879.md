# BFS代码

邻接表版：s为起始点

```cpp
// 定义图类
class Graph {
    int V; // 图的顶点数
    list<int> *adj; // 邻接表

public:
    Graph(int V); // 构造函数
    void addEdge(int v, int w); // 添加边
    void BFS(int s); // 从顶点 s 开始进行 BFS 搜索
};

void Graph::BFS(int s) {
    bool *visited = new bool[V]; // 访问标记数组
    for (int i = 0; i < V; i++)
        visited[i] = false;

    queue<int> queue; // 创建一个队列用于 BFS

    visited[s] = true; // 标记起始点为已访问
    queue.push(s); // 将起始点加入队列

    while (!queue.empty()) {
        s = queue.front(); // 获取队列前端元素
        cout << s << " "; // 打印当前顶点
        queue.pop(); // 从队列中移除元素

        // 获取所有邻接点
        for (auto i = adj[s].begin(); i != adj[s].end(); ++i) {
            if (!visited[*i]) { // 如果邻接点未被访问
                visited[*i] = true; // 标记为已访问
                queue.push(*i); // 加入队列
            }
        }
    }
}
```

    queue[++rear] = startVertex;

        int currentVertex = queue[front++];
        printf("%d ", currentVertex);

        for (int i = 0; i < graph->numVertices; i++) {
            if (graph->adjMatrix[currentVertex][i] && !visited[i]) {
                queue[++rear] = i;

            }
        }
    }
}
